'use strict';
 
const BluebirdPromise = require('bluebird');
const Cloudant = require('cloudant');
const cloudantCredential = require('../util/cred_cloudant').getCloudantCredentials('responses');
const logger = require('../util/logger');
 
const cloudant = Cloudant(cloudantCredential);
const VENUE_DB_NAME = 'venue_type_response';
const DEFAULT_PAGE_LIMIT = 10000;
 
let _db = BluebirdPromise.promisifyAll(cloudant.use(VENUE_DB_NAME));
 
const getType = (id) => {
  return _db.getAsync(id).then((body) => {
    return body;
  });
};
 
const getTypes = () => {
  return _db.listAsync({include_docs: true}).then(body => body.rows.filter(item => item.doc.type).map(item => item.doc));
};
 
const insertType = (venueType) => {
  return _db.insertAsync(venueType).then((doc) => _db.getAsync(doc.id));
};
 
const findType = (selector) => { // find document in cloudant by using selector
  return _db.findAsync({selector});
};
 
//delete caq document from cloudant
const deleteType = (id) => {
  return getType(id).then((result) => { // check if deleting faq is existing in cloudant
    logger.debug('result from getting doc : ' + JSON.stringify(result));
    let docId = result._id;
    let docRev = result._rev;
    let delDoc = result;
    return _db.destroyAsync(docId, docRev).then(() => {
      return delDoc;
    });
  });
};
 
const bulkTypes = (types) => {
  return _db.bulkAsync(types, {});
};
 
module.exports = {
  getType,
  getTypes,
  insertType,
  deleteType,
  findType,
  bulkTypes,
};
