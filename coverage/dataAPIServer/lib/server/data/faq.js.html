const Cloudant = require('cloudant');
const BluebirdPromise = require('bluebird');
const uuidV1 = require('uuid/v1');
 
const logger = require('../util/logger');
const cloudantCredential = require('../util/cred_cloudant').getCloudantCredentials('responses');
 
const FAQ_DB_NAME = 'faq';
 
const cloudant = Cloudant(cloudantCredential);
const _db = BluebirdPromise.promisifyAll(cloudant.use(FAQ_DB_NAME));
 
const insertFaq = (doc) => { //insert new document into cloudant
  doc._id = 'faq_' + uuidV1();
  return _db.insertAsync(doc);
};
 
//register faq document into cloudant
const addFaq = (params) => {
  let doc = {
    keywords: params.keywords,
    question: params.question,
    answer: params.answer,
    stt_training: params.stt_training
  };
  return insertFaq(doc);
};
 
const findFaq = (selector) => { // find document in cloudant by using selector
  return _db.findAsync({selector});
};
 
const getFaq = (id)  => { // find document in cloudant by using selector
  return _db.getAsync(id);
};
 
const getFaqs = () => {
  return _db.viewAsync('view_faqs', 'faqs', {include_docs: true});
};
 
const deleteFaq = (id) => {
  return getFaq(id).then((result) => { // check if deleting faq is existing in cloudant
    logger.debug('result from getting doc : ' + JSON.stringify(result));
    let docId = result._id;
    let docRev = result._rev;
    let delDoc = result;
    return _db.destroyAsync(docId, docRev).then(() => delDoc);
  }).catch((err) => {
    logger.debug(err);
    throw {result: 'there is no response'};
  });
};

const updateFaq = (orgDoc, params) => {
  let doc = {
    _id: orgDoc._id,
    _rev: orgDoc._rev,
    keywords: params.keywords,
