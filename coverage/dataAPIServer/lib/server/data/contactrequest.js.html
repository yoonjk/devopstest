'use strict';
const extend = require('util')._extend;
const Cloudant = require('cloudant');
const BluebirdPromise = require('bluebird');
const logger = require('../util/logger');
const cloudantCredential = require('../util/cred_cloudant').getCloudantCredentials('responses');
 
const CR_DB_NAME = 'contactrequests';
 
const cloudant = Cloudant(cloudantCredential);
const _db = BluebirdPromise.promisifyAll(cloudant.use(CR_DB_NAME));
 
const bulkUpdateRequests = (docs) => {
  return _db.bulkAsync({docs}, {all_or_nothing: true});
};
 
const createRequest = (doc) => {
  return _db.insertAsync(doc);
};
 
const queryRequests = (selector) => {
  return _db.findAsync({selector});
};
 
const getRequests = () => {
  return _db.listAsync({include_docs: true}).then(body => body.rows.filter(item => item.doc.request).map(item => item.doc));
};
 
const getRequest = (id) => {
  return _db.getAsync(id);
};
 
const updateRequest = (id, doc) => {
  return _db.getAsync(id).then((retDoc) => {
    doc._id = retDoc._id;
    doc._rev = retDoc._rev;
    return _db.insertAsync(doc);
  });
};
 
module.exports = {
  bulkUpdateRequests,
  createRequest,
  queryRequests,
  getRequests,
  getRequest,
  updateRequest,
};
