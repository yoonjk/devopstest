'use strict';
 
const winston = require('winston');
 
var logger = new winston.Logger({
  level: process.env.LOGGER_LEVEL || 'info',
  rewriters: [
    (level, msg, meta) => {
      return meta;
    }
  ],
  transports: [
    new (winston.transports.Console)({
      json: false,
      timestamp: true,
      colorize: true,
      label: process.env.APP_PREFIX || 'Data API server'
    })
  ]
});
 
module.exports = logger;
