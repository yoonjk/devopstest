const meridiemTimeTo24HourTime = (meridiem) => {
  let old = meridiem.toLowerCase();
  const mIndex = old.search(/am|pm/);
  let mr = 'am';
 
  if (mIndex !== -1) {
    mr = old.substr(mIndex, 2);
  }
 
  old = old.replace(mr, '').trim();
  const res = old.split(':');
  let hour = parseInt(res[0]);
  if (mr === 'pm') {
    hour += 12;
  }
 
  let retTime;
  if (res[1]) {
    retTime = hour + ':' + res[1];
  } else {
    retTime = hour + ':00';
  }
  return retTime;
};
