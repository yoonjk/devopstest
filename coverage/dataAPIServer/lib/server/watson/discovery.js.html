const getEnvId = (name, callback) => {
  return new BluebirdPromise((resolve, reject) => {
    if (envId) {
      resolve(envId);
    } else {
      discovery.getEnvironmentsAsync({}).then((result) => {
        let env = result.environments.find(environment => environment.name === name);
        if (env) {
          envId = env.environment_id;
          resolve(envId);
          return;
        }
        logger.debug('there is no env');
        discovery.createEnvironmentAsync({name, description: name, size: 1}).then((data) => {
          if (data) {
            envId = data.environment_id;
            resolve(envId);
          }
        }).catch((err) => reject(err));
      });
    }
  });
};
 
const getConfigurationId = (id) => {
  return new BluebirdPromise((resolve, reject) => {

const getCollectionId = (name, language, callback) => {
  return new BluebirdPromise((resolve, reject) => {
    if (collectionId) {
      resolve(collectionId);
    } else {
      getEnvId(discoveryEnvName).then((envId) => {
        discovery.getCollectionsAsync({environment_id: envId}).then((result) => {
          let col = result.collections.find(collection => collection.name === name);
          if (col) {
            collectionId = col.collection_id;
            resolve(collectionId);
            return;
          }
          logger.debug('there is no collection');
          getConfigurationId(envId).then((configId) => {
            discovery.createCollectionAsync({environment_id: envId, description: name, configuration_id: configId, name, language}).then((data) => {
              if (data) {
                collectionId = data.collection_id;
                resolve(collectionId);
              }
            });
          });
        }).catch((err) => reject(err));
      });
    }
  });
};
