const updateReservation = (req, res) => {
  if (req.body.user.length > 0) {
    reservationDao.getUserReservation(req.body.user, req.body.venue).then((reservations) => {
      if (reservations.length === 0) {
        res.status(404).json({error: 'There is no reservations to update'});
      } else if (reservations.length > 1) {
        res.status(400).json({error: 'There are multiple reservations'});
      } else {
        const newReservation = reservations[0];
        newReservation.request.venue = req.body.venue;
        newReservation.request.date = req.body.date;
        newReservation.request.time = req.body.time;
        newReservation.request.count = req.body.people;
        let action;
        if (parseInt(req.body.time.substr(0, 2)) > 15) {
          action = 'dinner';
        } else {
          if (parseInt(req.body.time.substr(0, 2)) > 11) {
            action = 'lunch';
          } else {
            action = 'breakfast';
          }
        }
        newReservation.request.action = action;
        reservationDao.updateReservation(newReservation).then(updatedReservation => {
          res.json(updatedReservation);
        });
      }
    })
    .catch((err) => {
      logger.error(err);
      res.status(500).json({error: err.message});
    });
  } else {
    res.status(400).json({error: 'Invalid data'});
  }
};
